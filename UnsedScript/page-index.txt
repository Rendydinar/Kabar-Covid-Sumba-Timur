  createStyles,
  Divider,
  makeStyles,
  Theme,
  Typography,
} from '@material-ui/core';

import Head from 'next/head';
import React, { ReactElement, useEffect, useState } from 'react';
import Card from '../components/Card';
import DataCovidPerkecamatan from '../components/DataCovidPerkecamatan';
import Header from '../components/Header';
import Jumbotron from '../components/Jumbotron';
import Layout from '../components/Layout';
import Top10Kecamatan from '../components/Top10Kecamatan';
import { getCovidData } from '../fetchData/getCovidData';
import { getCovidDataPerkecamatan } from '../fetchData/getCovidDataPerkecamatan';
import { IKecamatan, IKelurahan } from '../interfaces';
import { getHariIni } from '../utils/date';

  createStyles({
    sliderContainer: {
      height: '80vh',
    },
    typeMonitoring: {
      fontSize: '26px',
      fontWeight: 700,

    wrapper: {

      },
    },
    containerTypeMonitoring: {
      padding: '0 125px',
      [theme.breakpoints.down('md')]: {
        padding: '0 50px',
      },
      [theme.breakpoints.down('sm')]: {
        padding: 0,
      },
    },
    monitoringTitle: {
      fontSize: '20px',
      fontWeight: 700,
      textTransform: 'uppercase',

      display: 'flex',

    divider: {

interface IProps {
  data: any;
}

const images = [
  'https://images.pexels.com/photos/3740695/pexels-photo-3740695.jpeg?auto=compress&cs=tinysrgb&dpr=2&h=750&w=1260',
  'https://images.pexels.com/photos/3740446/pexels-photo-3740446.jpeg?auto=compress&cs=tinysrgb&dpr=2&h=750&w=1260',
  'https://images.pexels.com/photos/351265/pexels-photo-351265.jpeg?auto=compress&cs=tinysrgb&dpr=2&h=750&w=1260',
  'https://images.pexels.com/photos/924675/pexels-photo-924675.jpeg?auto=compress&cs=tinysrgb&dpr=2&h=750&w=1260',
];

const imageStyle = (src: string) => ({
  backgroundSize: 'cover',
  backgroundImage: `url(${src})`,
  height: '100%',
  width: '100%',
});
interface IData {
  success: boolean;
  data: {
    data_covid_sumba_timur: any;
    data_covid_perkecamatan_sumba_timur: {
      date: string;
      data: IKecamatan[];
    };
    top_10_kelurahan_kasus_covid: IKelurahan[];
  };
}

const Home: React.FC<IProps> = (): ReactElement => {
  const classes = useStyles();
  const [data, setData] = useState<IData>();
  const [loading, setLoading] = useState<boolean>(false);

  const getData = async () => {
    setLoading(true);
    try {
      const responseGetCovidData: any = await getCovidData();
      const responseGetCovidDataPerkecamatan: any =
        await getCovidDataPerkecamatan();
      let listkelurahan: IKelurahan[] = [];
      responseGetCovidDataPerkecamatan.data.map((kecamatan: IKecamatan) => {
        kecamatan.kelurahan.map((kelurahan) => {
          listkelurahan.push(kelurahan);
        });
      });
      listkelurahan = sortBy(listkelurahan, 'total').reverse();
      setData({
        success: true,
        data: {
          data_covid_sumba_timur: responseGetCovidData,
          data_covid_perkecamatan_sumba_timur: responseGetCovidDataPerkecamatan,
          top_10_kelurahan_kasus_covid: listkelurahan.slice(0, 10),
        },
      });
    } catch (err) {
      setData({
        success: false,
        data: {
          data_covid_sumba_timur: {},
          data_covid_perkecamatan_sumba_timur: {
            date: '',
            data: [],
          },
          top_10_kelurahan_kasus_covid: [],
        },
      });
    }
    setLoading(false);
  };

  useEffect(() => {
    async function asyncFuncDefault() {
      await getData();
    }
    asyncFuncDefault();
  }, []);

  return (
    <Layout>
      <Head>
        <title>Kabar Covid Sumba Timur</title>
        <link rel='icon' href='/favicon.ico' />
      </Head>
      <Header />
      <div className={classes.sliderContainer}>
        <Slider activeIndex={1} slidesAtOnce={1} auto={6000} hasBullets>
          {images.map((image, index) => (
            <div key={index} draggable='false' style={imageStyle(image)} />
          ))}
        </Slider>
      </div>
      <div>
        <Jumbotron
          title='Data Monitoring Harian'
          description={`Update: ${getHariIni()}`}
        />
        {/* Sampel Dikirim / Diperiksa */}
        <div className={classes.wrapper}>
          <Typography className={classes.typeMonitoring}>
            Sampel Dikirim / Diperiksa
          </Typography>
          {/* TCM / PCR */}
          <div className={classes.containerTypeMonitoring}>
            <Typography className={classes.monitoringTitle}>
              TCM / PCR
            </Typography>
            <div className={classes.monitoringSectionContainer}>
              <Card
                icon='ðŸ˜Ÿ'
                title='Positif'
                total={
                  loading
                    ? data?.data.data_covid_sumba_timur?.data?.tcm_pcr
                        ?.positif ?? 0
                    : 0
                }
              />
              <Card
                icon='ðŸ˜Š'
                title='Negatif'
                total={
                  data?.data?.data_covid_sumba_timur?.data?.tcm_pcr?.negatif ??
                  0
                }
              />
              <Card
                title='Total'
                total={
                  data?.data?.data_covid_sumba_timur?.data?.tcm_pcr?.positif ??
                  0 +
                    data?.data?.data_covid_sumba_timur?.data?.tcm_pcr
                      ?.negatif ??
                  0
                }
              />
            </div>
          </div>
          {/* Rapid Antigen */}
          <div className={classes.containerTypeMonitoring}>
            <Typography className={classes.monitoringTitle}>
              Rapid Antigen
            </Typography>
            <div className={classes.monitoringSectionContainer}>
              <Card
                icon='ðŸ˜Ÿ'
                title='Positif'
                total={
                  data?.data?.data_covid_sumba_timur?.data?.rapid_antigen
                    ?.positif ?? 0
                }
              />
              <Card
                icon='ðŸ˜Š'
                title='Negatif'
                total={
                  data?.data?.data_covid_sumba_timur?.data?.rapid_antigen
                    ?.negatif ?? 0
                }
              />
              <Card
                title='Total'
                total={
                  data?.data?.data_covid_sumba_timur?.data?.rapid_antigen
                    ?.positif ??
                  0 +
                    data?.data?.data_covid_sumba_timur?.data?.rapid_antigen
                      ?.negatif ??
                  0
                }
              />
            </div>
          </div>
        </div>
        <Divider className={classes.divider} />

        {/* Positif Covid-19 */}
        <div
          style={{
            marginTop: '15px',
            padding: '0 30px',
            // paddingBottom: '50pc',
          }}
        >
          <Typography className={classes.typeMonitoring}>
            Positif Covid-19
          </Typography>
          <div className={classes.containerTypeMonitoring}>
            <Typography className={classes.monitoringTitle}>
              Penambahan Kasus Hari Ini
            </Typography>
            <div className={classes.monitoringSectionContainer}>
              <Card
                title='Antigen'
                total={
                  data?.data?.data_covid_sumba_timur?.data?.positif_covid
                    ?.penambahan_kasus_harian?.antigen ?? 0
                }
              />
              <Card
                title='PCR / TCM'
                total={
                  data?.data?.data_covid_sumba_timur?.data?.positif_covid
                    ?.penambahan_kasus_harian?.pcr_tcm ?? 0
                }
              />
            </div>
          </div>
          <div className={classes.containerTypeMonitoring}>
            <Typography className={classes.monitoringTitle}>
              Data Total Positif
            </Typography>
            <div className={classes.monitoringSectionContainer}>
              <Card
                icon='ðŸ˜Ÿ'
                title='Total Positif'
                total={
                  data?.data?.data_covid_sumba_timur?.data?.positif_covid
                    ?.total_positif ?? 0
                }
              />
              <Card
                icon='ðŸ¤’'
                title='Total Dirawat'
                total={
                  data?.data?.data_covid_sumba_timur?.data?.positif_covid
                    ?.total_dirawat ?? 0
                }
              />
              <Card
                icon='ðŸ˜ƒ'
                title='Total Sembuh'
                total={
                  data?.data?.data_covid_sumba_timur?.data?.positif_covid
                    ?.total_sembuh ?? 0
                }
              />
              <Card
                icon='ðŸ˜¢'
                title='Total Meninggal'
                total={
                  data?.data?.data_covid_sumba_timur?.data?.positif_covid
                    ?.total_meninggal ?? 0
                }
              />
            </div>
          </div>
        </div>
      </div>

      <DataCovidPerkecamatan
        dataPerkecamatan={
          data?.data?.data_covid_perkecamatan_sumba_timur ?? {
            date: '',
            data: [],
          }
        }
      />
      <Top10Kecamatan
        top10Kelurahan={data?.data?.top_10_kelurahan_kasus_covid ?? []}
      />
    </Layout>
  );
};

export default Home;
